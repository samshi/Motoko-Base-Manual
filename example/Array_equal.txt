public func equal(a : [A], b : [A], eq : (A, A) -> Bool) : Bool {
  if (a.size() != b.size()) {
    return false;
  };
  var i = 0;
  while (i < a.size()) {
    if (not eq(a[i], b[i])) {
      return false;
    };
    i += 1;
  };
  return true;
}
